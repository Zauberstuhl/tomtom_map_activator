# TomTom Map Activator for Linux

Usage:

    # will list all detected cards
    # on your tomtom device
    $ ./activate.sh /mnt/tomtom
    0) Iberia/Iberia
    
    # this will activated the
    # first card in the list
    ./activate.sh /mnt/tomtom 0
